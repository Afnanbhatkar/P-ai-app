# === FINAL BULLETPROOF CHAT SYSTEM (NEVER CRASHES) ===
CHAT_FILE = "chats.json"

def load_chats():
    if not os.path.exists(CHAT_FILE):
        return []
    try:
        with open(CHAT_FILE, "r", encoding="utf-8") as f:
            data = f.read().strip()
            if not data:
                return []
            return json.loads(data)
    except (json.JSONDecodeError, Exception) as e:
        print(f"chats.json corrupted! Starting fresh... ({e})")
        return []  # Never crash â€” just reset

def save_chats(chats):
    try:
        with open(CHAT_FILE, "w", encoding="utf-8") as f:
            json.dump(chats, f, indent=2, ensure_ascii=False)
    except Exception as e:
        print(f"Failed to save chats: {e}")

def create_new_chat(first_message="New Chat"):
    chats = load_chats()
    chat_id = len(chats) + 1
    title = first_message.strip()
    if len(title) > 40:
        title = title[:37] + "..."
    if not title or title.lower() in ["hi", "hello", "hey", "jarvis", "new chat"]:
        title = f"Chat {chat_id}"
    
    new_chat = {
        "id": chat_id,
        "title": title,
        "messages": [{"role": "user", "content": first_message}]
    }
    chats.append(new_chat)
    save_chats(chats)
    return new_chat

def add_message_to_chat(chat_id, role, content):
    # Make sure chat_id is integer
    try:
        chat_id = int(chat_id)
    except:
        return None

    chats = load_chats()
    for chat in chats:
        if chat["id"] == chat_id:  # Now 1 == 1 â†’ True (not "1" == 1 â†’ False)
            chat["messages"].append({"role": role, "content": content})
            
            # Update title after first reply
            if len(chat["messages"]) == 2 and role == "assistant":
                first_user_msg = chat["messages"][0]["content"]
                new_title = first_user_msg.strip()
                if len(new_title) > 40:
                    new_title = new_title[:37] + "..."
                if new_title and new_title.lower() not in ["hi", "hello", "hey", "jarvis"]:
                    chat["title"] = new_title
                    
            save_chats(chats)
            return chat
    return None

# Function to interact with PIAI website using Selenium
def interact_with_piai(query):
    # Check if the query is about the current time, open youtube, or other predefined actions
    if "current time" in query.lower():
        current_time = time.strftime("%H:%M:%S", time.localtime())
        return f"The current time is {current_time}."
    elif "open youtube" in query.lower():
        return webbrowser.open("https://www.youtube.com/")
    elif 'google search' in query:
        # Perform Google search
        import wikipedia as googlescrap
        query = query.replace('search', '')
        query = query.replace('google', '')
        try:
            # Retrieve Wikipedia summary
            summary = googlescrap.summary(query, sentences=3)
            return summary
        except Exception as e:
            return "Wikipedia summary not available."
    elif 'visit' in query or '.org' in query:
        # Open the requested website
        query = query.replace('website', '')
        query = query.replace('open', '')
        query = query.replace('jarvis', '')
        query = query.replace('visit', '')
        query = query.replace(' ', '')
        query = query.replace('www.', '')
        webbrowser.open('www.' + query + '.com')
        return "Visiting " + query
    elif 'you need a break' in query:
        keyboard.press_and_release("ctrl + w")
        return "bye !!"

@app.route('/api/ai', methods=['POST'])
def get_ai_response():
    data = request.get_json(silent=True) or {}
    message = data.get("message", "").strip()
    chat_id = data.get("chatId")  # â† This comes as STRING from React ("1", "2", etc.)

    if not message or not chat_id:
        return jsonify({"error": "Missing data"}), 400

    # THIS IS THE MAGIC FIX â€” CONVERT STRING TO INTEGER
    try:
        chat_id = int(chat_id)  # â† "1" â†’ 1   |   "5" â†’ 5
    except (ValueError, TypeError):
        return jsonify({"error": "Invalid chatId"}), 400

    # Now it will find the correct chat even after refresh
    chat = add_message_to_chat(chat_id, "user", message)
    if not chat:
        return jsonify({"error": "Chat not found"}), 404

    response = interact_with_piai(message)
    add_message_to_chat(chat_id, "assistant", response)

    return jsonify({"message": response})
@app.route('/api/new-chat', methods=['POST'])
def new_chat():
    data = request.get_json(silent=True) or {}
    first_message = data.get("firstMessage", "Hi").strip()

    reply = interact_with_piai(first_message)
    add_message_to_chat(chat["id"], "assistant", reply)
    
    return jsonify({"chat": chat, "response": reply})
